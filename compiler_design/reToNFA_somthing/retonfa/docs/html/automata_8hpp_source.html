266Automata processRegEx(const string& exp)
267{
268 std::stack<Automata> s;
269 for(const char& alphabet : exp)
270 {
271 if(alphabet == '*')
272 {
273 Automata a = s.top();
274 s.pop();
275 a = closure(a);
276 s.push(a);
277 }
278 else if(alphabet == '.')

283 s.pop();
284
285 s.push(concatenation(first,second));
286 }
287 else if(alphabet == '|')

290 s.pop();
291 Automata first = s.top();
292 s.pop();
293
294 s.push(_union(first,second));
295 }
296 else
297 {
298 s.emplace(alphabet);
299 }
300 }

386Automata convertToDFA(const StateSharedPointer& nfa_start_state,const std::vector<char> alphabets)
387{
388 Automata dfa;
389 dfa.initial_state = std::make_shared<State>(*nfa_start_state);
390 dfa.initial_state->id = ++id_counter;
391
392 TransitionSetShared states;
393 states.insert(dfa.initial_state);

395 std::queue<StateSharedPointer> q;
396 q.push(dfa.initial_state);

398 while(not q.empty())
399 {
400 StateSharedPointer current_state = q.front();
401 q.pop();
402
403 TransitionSet closure_set;
404 if(current_state->combined_set)
405 {
406 closure_set = epsilonClosure(current_state->combined_set.value());
407 }
408 else

411 }
412
413 auto accepting_itr = std::find_if(closure_set.begin(),closure_set.end(),[](const StateWeakPointer& s){
414 if(StateSharedPointer(s)->is_accepting)
415 {
416 return true;
417 }
418 else return false;
419 });
420
421 if(accepting_itr!=closure_set.end())
422 {
423 current_state->is_accepting = true;
424 }
425
426 for(size_t i=0;i<alphabets.size();i++)
427 {
428 TransitionSet result_set = epsilonClosure(move(closure_set,alphabets.at(i)));
429
430 if(result_set.empty()) continue;
431
432 StateSharedPointer combined_state = std::make_shared<State>(++id_counter);
433 combined_state->combined_id = makeCombinedId(result_set);
434 combined_state->combined_set = std::make_optional(result_set);
435
436 if(auto state_itr = states.find(combined_state); state_itr!=states.end())
437 {
438 combined_state = StateSharedPointer(*state_itr);
439 --id_counter;
440 }
441
442 if(current_state == dfa.initial_state)
443 {
444 current_state->transitions.erase(alphabets.at(i));
445 }
446 current_state->insertTransition(alphabets.at(i),combined_state);
447
448 if(states.insert(combined_state).second)
449 {
450 q.push(combined_state);
451 }
452 }
453
454 if(current_state == dfa.initial_state)
455 {
456 current_state->transitions.erase(epsilon);
457 }
458 }
459
460 dfa.states.insert(states.begin(),states.end());
